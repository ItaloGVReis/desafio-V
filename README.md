# Clonar o repositório
`git clone https://github.com/ItaloGVReis/desafio-V.git`

# Inicializar o node
`npm init -y`

# Instalar o Express.js
`npm install express`

# Instalar o mysql
`npm install mysql`

# Iniciar o servidor Node.js
`node app.js`

# Acessar a página
`http://localhost:3000`

# Cada alteração no código, rodar novamente no terminal
`node app.js`
